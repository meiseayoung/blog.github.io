点击执行方法一回到顶部
点击执行方法二回到顶部
点击执行方法三回到顶部

方法一

        var Location=window.location+"#";//这里不能把变量写到事件函数里面，那样的话每次都会在地址后面添加"#"
        $(".backTop").click(function(){
            window.location=Location;
        })

方法二

代码由 M提供 
        $(".backTop").click(function(){
           $(document).scrollTop(0);
        })

方法三

带动画效果 
         $(".backTop").click(function(e){
            $("html,body").animate({scrollTop: (e.pageY-$(this).offset().top)}, 1000);
         })
